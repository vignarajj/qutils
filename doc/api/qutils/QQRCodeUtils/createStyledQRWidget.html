static Widget createStyledQRWidget(
  String data, {
  double size = 200.0,
  Color foregroundColor = Colors.black,
  Color backgroundColor = Colors.white,
  Widget? embeddedImage,
  double embeddedImageSizePercent = 0.25,
  BorderRadius? borderRadius,
  EdgeInsets padding = const EdgeInsets.all(10.0),
  BoxShadow? shadow,
}) {
  Widget qrWidget = QrImageView(
    data: data,
    version: QrVersions.auto,
    size: size,
    gapless: false,
    padding: padding,
    backgroundColor: backgroundColor,
    embeddedImage:
        null, // This would need actual image handling with proper asset path
    embeddedImageStyle: embeddedImage != null
        ? QrEmbeddedImageStyle(
            size: Size.square(size * embeddedImageSizePercent),
          )
        : null,
  );

  if (borderRadius != null || shadow != null) {
    qrWidget = Container(
      decoration: BoxDecoration(
        borderRadius: borderRadius,
        boxShadow: shadow != null ? [shadow] : null,
      ),
      child: ClipRRect(
        borderRadius: borderRadius ?? BorderRadius.zero,
        child: qrWidget,
      ),
    );
  }

  return qrWidget;
}
